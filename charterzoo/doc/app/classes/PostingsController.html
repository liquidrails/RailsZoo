259: def process_chunk(chunk)
260:       case chunk
261:       when /^-/
262:         if chunk.length  1
263:           [:not]
264:         else
265:           [:not, *process_chunk(chunk[1..-1])]
266:         end
267:       when /^\(.*\)$/
268:         if chunk.length  2
269:           [:left_paren, :right_paren]
270:        else          
271: [:left_paren].concat(process_chunk(chunk[1..-2])) << :right_paren
272:         end
273:       when /^\(/
274:         if chunk.length  1
275:           [:left_paren]
276:         else
277:           [:left_paren].concat(process_chunk(chunk[1..-1]))
278:         end
279:       when /\)$/
280:         if chunk.length  1
281:           [:right_paren]
282:         else
283:           process_chunk(chunk[0..-2]) << :right_paren
284:         end
285:       when 'and'
286:         [:and]
287:       when 'or'
288:         [:or]
289:       when 'not'
290:         [:not]
291:       else
292:         [chunk]
293:       end
